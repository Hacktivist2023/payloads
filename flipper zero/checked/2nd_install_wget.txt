REM and installed

REM -------------------------------------------------------
REM Close all Windows
WINDOWS d
DELAY 200
REM Enter commandline
WINDOWS r
DELAY 200
REM Start powershell
STRING powershell Start-Process powershell -Verb runAs
ENTER
DELAY 4000
LEFTARROW
DELAY 1000
ENTER
DELAY 1000
REM Install wget via winget
STRING winget install wget
DELAY 500
ENTER
DELAY 1000
STRING exit
ENTER
